

document.addEventListener('DOMContentLoaded', function() {
    fetch('./database/atividades.json')
        .then(response => response.json())
        .then(data => {
            //**carregar a primeira pergunta do meu json */
            const perguntaData = data[0];

            document.getElementById('pergunta').textContent = perguntaData.pergunta;

            document.getElementById('imagem-principal').src = perguntaData.imagem;

            document.getElementById('opcao-1-img').src = perguntaData.opcoes[0].resposta;
            document.getElementById('opcao-2-img').src = perguntaData.opcoes[1].resposta;
        })
        //**criar imagem que substitua o do meu banco caso não ela não carregue */
        .catch(error => console.error('Erro ao carregar o JSON:', error));
});
